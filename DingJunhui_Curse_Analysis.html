# now define a function to get this organized results for each tournament given url
def get_tour_results(tour_url):
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/122.0.0.0 Safari/537.36"
    }
    response = requests.get(tour_url, headers=headers)

    if response.status_code == 200:
        soup = BeautifulSoup(response.text, "html.parser")
        # Find all tables on the page
        tables = soup.find_all("table")
        # Extract match results (assuming results are in the first table)
        if tables:
            tournament_results = []
            for table in tables:
                rows = table.find_all("tr")
                match_results = []
                for row in rows:
                    cols = row.find_all("td")
                    # cols is all the <td> elements in the row but class does not start with "even info"
                    cols = [ele for ele in cols if not ele.get('class') or not ele.get('class')[0].startswith("info")]
                    cols = [ele.text.strip() for ele in cols]
                    match_results.append([ele for ele in cols if ele])  # Get rid of empty values
                tournament_results.append(match_results)
        else:
            print("No tables found on the page.")
            return []
        # get rid of empty lists in tournament_results
        organized_results = []
        for result in tournament_results:
            organized_result = []
            for match in result:
                if match:  # Only add non-empty matches
                    organized_result.append(match)
                else:
                    continue  # Stop adding on first empty match
            if organized_result:  # Only add non-empty organized results
                organized_results.append(organized_result)
        return organized_results
    else:
        print(f"Failed to retrieve page, status code: {response.status_code}")
        return []

# record curse situation
curse_work_count = defaultdict(int)
tour_app_count = defaultdict(int)
for i in range(len(ordered_tournaments)):
    tournament_name = ordered_tournaments[i]
    tournament_results = simplified_results[tournament_name]
    winner = tournament_results[0][0] if tournament_results else None
    runner_up = tournament_results[0][1] if tournament_results and len(tournament_results[0]) > 1 else None
    if winner:
        print(f"The winner of {tournament_name} is {winner}.")
    else:
        print(f"No results found for {tournament_name}.")
    appearance_count = defaultdict(int)
    last_win_against = {}
    for match in tournament_results:
        if len(match) == 2:
            player1, player2 = match[0], match[1]
            appearance_count[player1] += 1
            appearance_count[player2] += 1
            if appearance_count[player1] == 2:
                last_win_against[player1] = player2
    # check if winner lost in next tournament
    if i < len(ordered_tournaments) - 1:
        next_tournament_name = ordered_tournaments[i + 1]
        next_tournament_results = simplified_results[next_tournament_name]
        # lost means only appear in one match
        count = 0
        for match in next_tournament_results:
            if winner in match:
                count += 1
        #print(f"{winner}, {count}, {runner_up}")
        if count == 1:
            last_win_against[winner] = runner_up
    # delete winner
    del appearance_count[winner]
    # update
    for player, against in last_win_against.items():
        curse_work_count[against] += 1
    for player in appearance_count:
        tour_app_count[player] += 1
